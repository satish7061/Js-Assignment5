Qu 10. Write a program to fin min and max value in an array?

    function arrayMin(arr) {

        return arr.reduce((x, y) => ( x < y ? x : y ));
       
       }
       
       function arrayMax(arr) {
         
         
          return arr.reduce((x, y) => ( x > y ? x : y ));
       
       
       }

Qu 11. Write a program for Birthday Game?

    function Birthday_Game(arr,D ,M) {

        let count = 0;
        
         for(let i = 0; i < arr.length - 1; i++){
           
               let subArray = arr.slice(i, i + M);
               
               if (subArray.length === M) {
                 
                  let sum = subArray.reduce((a,b) => a + b, 0);
                  
                  if(sum === D){
                    
                       count += 1;
                   }
                }
          }
          return count;
     
     }
